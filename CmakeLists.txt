cmake_minimum_required(VERSION 3.16.3...3.19.7 FATAL_ERROR)

# Project Name
project(IVIMFitSlicer)

#-----------------------------------------------------------------------------
# Extension Metadata
#-----------------------------------------------------------------------------
set(EXTENSION_HOMEPAGE "https://github.com/Atakanisik/SlicerIVIMFit")
set(EXTENSION_CATEGORY "Diffusion")
set(EXTENSION_CONTRIBUTORS "Atakan Isik (Baskent University)")

set(EXTENSION_ICONURL "https://raw.githubusercontent.com/Atakanisik/SlicerIVIMFit/main/IVIMFitSlicer/Resources/Icons/IVIMFitSlicer.png")
set(EXTENSION_SCREENSHOTURLS "https://raw.githubusercontent.com/Atakanisik/SlicerIVIMFit/main/Screenshots/MainInterface.png" "https://raw.githubusercontent.com/Atakanisik/SlicerIVIMFit/main/Screenshots/Map_Plot_scree.png")
set(EXTENSION_DEPENDS "NA")

#-----------------------------------------------------------------------------
# Dependencies
#-----------------------------------------------------------------------------
find_package(Slicer REQUIRED)
include(${Slicer_USE_FILE})

#-----------------------------------------------------------------------------
# Modules
#-----------------------------------------------------------------------------

add_subdirectory(IVIMFitSlicer)


include(${Slicer_EXTENSION_GENERATE_CONFIG})

include(${Slicer_EXTENSION_CPACK})
